
DROP TABLE DeliveryOrder;
CREATE TABLE DeliveryOrder(
    DeliveryOrderId int NOT NULL AUTO_INCREMENT,
    CustName varchar(30),
    CustID int(4),
    CustAddress varchar(30),
    PublicationName varchar(30),
    DeliveryDate varchar(12),     
    PRIMARY KEY (DeliveryOrderId)
);
