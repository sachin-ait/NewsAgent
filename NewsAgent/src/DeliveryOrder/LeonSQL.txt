DROP TABLE Customer;
CREATE TABLE Customer (
    CustID int NOT NULL AUTO_INCREMENT,
    Name varchar(25),
    Address varchar(25),
    PhoneNumber int,
    DeliveryOrderID int,
    Payment int,
    Area varchar(25),    
    PRIMARY KEY (CustID)
);

ALTER TABLE Customer
ADD FOREIGN KEY (DeliveryOrderID)REFERENCES DeliveryOrder(DeliveryOrderID);
